010package org.kew.rmf.transformers;
011
012/**
013 * If the input matches the supplied filter, transform using the wrapped transformer.
014 * The filter can be negated by setting positiveFilter to false.
015 */
016public class FilteredTransformer implements Transformer{
017
018        private String filter;
019        private boolean positiveFilter = true;
020        private Transformer transformer;
021        
022        @Override
023        public String transform(String s) throws TransformationException {
024                boolean doTransform = s.matches(filter);
025                if (!positiveFilter)
026                        doTransform = !doTransform;
027                if (doTransform){
028                        return transformer.transform(s);
029                }
030                else
031                        return s;
032        }
033
034        public String getFilter() {
035                return filter;
036        }
037
038        public void setFilter(String filter) {
039                this.filter = filter;
040        }
041
042        public Transformer getTransformer() {
043                return transformer;
044        }
045
046        public void setTransformer(Transformer transformer) {
047                this.transformer = transformer;
048        }
049
050        public boolean isPositiveFilter() {
051                return positiveFilter;
052        }
053
054        public void setPositiveFilter(boolean positiveFilter) {
055                this.positiveFilter = positiveFilter;
056        }
057
058}
