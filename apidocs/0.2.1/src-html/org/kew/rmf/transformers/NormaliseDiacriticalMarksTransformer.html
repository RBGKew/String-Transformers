010package org.kew.rmf.transformers;
011
012import java.text.Normalizer;
013import java.util.HashMap;
014import java.util.Map;
015import java.util.regex.Pattern;
016
017/**
018 * Removes diacritical marks from letters, intended for dealing with OCR errors
019 * and language-ignorant string comparisons.
020 * <br/>
021 * Converts a string to its normalised form using unicode's NFKD form
022 * (http://unicode.org/reports/tr15/).  Translates some non-English letters
023 * to what an English speaker might consider an unaccented letter.
024 */
025public class NormaliseDiacriticalMarksTransformer implements Transformer {
026
027        /*
028         * See http://www.regular-expressions.info/unicode.html for Unicode character classes.
029         * See also Lucene's ASCIIFoldingFilter, which deals with a far greater range of characters.
030         */
031
032        private final Pattern accents = Pattern.compile("[\\p{M}]");
033
034        private final Map<String,String> additionalReplacements;
035        public NormaliseDiacriticalMarksTransformer() {
036                additionalReplacements = new HashMap<>();
037                additionalReplacements.put("Æ", "AE");
038                additionalReplacements.put("æ", "ae");
039                additionalReplacements.put("Œ", "OE");
040                additionalReplacements.put("œ", "oe");
041                additionalReplacements.put("Ø", "O");
042                additionalReplacements.put("ø", "o");
043                additionalReplacements.put("Ł", "L");
044                additionalReplacements.put("ł", "l");
045                additionalReplacements.put("Đ", "D");
046                additionalReplacements.put("đ", "d");
047                additionalReplacements.put("—", "-"); // emdash to hyphen
048                additionalReplacements.put("–", "-"); // ndash to hyphen
049        };
050
051        @Override
052        public String transform(String s) {
053                s = Normalizer.normalize(s, Normalizer.Form.NFKD);
054                for (String key : additionalReplacements.keySet()) {
055                        s = s.replace(key, additionalReplacements.get(key));
056                }
057                return accents.matcher(s).replaceAll("");
058        }
059}
