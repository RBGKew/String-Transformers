010package org.kew.rmf.transformers.authors;
011
012import java.util.regex.Pattern;
013
014import org.kew.rmf.transformers.Transformer;
015
016/**
017 * This transformer tries to identify all surnames of plant name authors in a string
018 * and deletes everything else.
019 */
020/*
021 * For examples see {@link org.kew.rmf.transformers.authors.SurnameExtractorTest}
022 */
023public class SurnameExtractor implements Transformer {
024
025        private Pattern linnaeusRemoveDot = Pattern.compile("L\\.(?=\\s)");
026        private Pattern surnameExtractorPattern = Pattern.compile("(?<!\\p{L})(-*\\p{L}\\.\\s*)(?=[^$\\)])");
027        private Pattern linnaeusAddDot = Pattern.compile("L ");
028
029        @Override
030        public String transform(String s) {
031                // Linnaeus special: first we remove the Dot after "L" in order to make it appear as a surname,
032                //  but only where it's very likely to be Linnaeus; afterwards we'll add it again.
033                s = linnaeusRemoveDot.matcher(s).replaceAll("L ");
034
035                s = surnameExtractorPattern.matcher(s).replaceAll("");
036
037                s = linnaeusAddDot.matcher(s).replaceAll("L.");
038                return s;
039        }
040}
