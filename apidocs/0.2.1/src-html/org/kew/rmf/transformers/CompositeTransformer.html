010package org.kew.rmf.transformers;
011
012import java.util.ArrayList;
013import java.util.List;
014
015/**
016 * A CompositeTransformer can have other transformers and
017 * executes them in order on the given string.
018 */
019public class CompositeTransformer implements Transformer{
020
021        private List<Transformer> transformers = new ArrayList<>();
022
023        @Override
024        public String transform(String s) throws TransformationException {
025                for (Transformer t : transformers) {
026                        s = t.transform(s);
027                }
028                return s;
029        }
030
031        public List<Transformer> getTransformers() {
032                return transformers;
033        }
034
035        public void setTransformers(List<Transformer> transformers) {
036                this.transformers = transformers;
037        }
038
039        public void addTransformer(Transformer transformer) {
040                transformers.add(transformer);
041        }
042
043}
