010package org.kew.rmf.transformers;
011
012import java.util.regex.Pattern;
013
014/**
015 * This is a chain of three transformations:
016 * <ol>
017 *      <li>replace diacritical characters with their Latin equivalent ({@link NormaliseDiacriticalMarksTransformer})</li>
018 *      <li>replace all non-alphanumeric characters with {@link #replacement} (default: space)</li>
019 *      <li>replace multiple whitespace occurrences with one whitespace</li>
020 * </ul>
021 * Returns a trimmed result.
022 * <br/>
023 * This transformer works with Unicode text, unlike {@link StripNonAsciiAlphanumericCharactersTransformer}.
024 */
025public class StripNonAlphanumericCharactersTransformer implements Transformer {
026
027        private NormaliseDiacriticalMarksTransformer normaliseDiacriticalMarksTransformer = new NormaliseDiacriticalMarksTransformer();
028        private SqueezeWhitespaceTransformer squeezeWhitespaceTransformer = new SqueezeWhitespaceTransformer();
029
030        private final Pattern nonAlphanumeric = Pattern.compile("[^\\p{L}\\p{N}]");
031        private String replacement = " ";
032
033        @Override
034        public String transform(String s) {
035                s = normaliseDiacriticalMarksTransformer.transform(s);
036                s = nonAlphanumeric.matcher(s).replaceAll(replacement);
037                s = squeezeWhitespaceTransformer.transform(s);
038                return s.trim();
039        }
040
041        public String getReplacement() {
042                return replacement;
043        }
044        public void setReplacement(String replacement) {
045                this.replacement = replacement;
046        }
047}
