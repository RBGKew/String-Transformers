010package org.kew.rmf.transformers;
011
012import java.util.regex.Matcher;
013import java.util.regex.Pattern;
014
015/**
016 * A generic transformer that extracts all occurrences of a pattern (regEx)
017 * in a string
018 * <br/>
019 * It takes two optional parameters, {@link #removeMultipleWhitespaces} (default true) and
020 * {@link #trimIt} (default true)
021 */
022public class RegexExtractorTransformer implements Transformer {
023
024        private Pattern regex;
025        private boolean removeMultipleWhitespaces = true;
026        private boolean trimIt = true;
027
028        @Override
029        public String transform(String s) {
030                StringBuilder sb = new StringBuilder();
031                if (s != null){
032                        if (this.removeMultipleWhitespaces) {
033                                s = SqueezeWhitespaceTransformer.MULTIPLE_WHITESPACE.matcher(s).replaceAll(" ");
034                        }
035                        if (this.trimIt) {
036                                s = s.trim();
037                        }
038
039                        Matcher m = regex.matcher(s);
040                        while (m.find()){
041                                if (sb.length() > 0) sb.append(" ");
042                                sb.append(m.group());
043                        }
044                }
045                return sb.toString();
046        }
047
048        public String getRegex() {
049                return regex.pattern();
050        }
051        public void setRegex(String regex) {
052                this.regex = Pattern.compile("(" + regex + ")");
053        }
054
055        public boolean isRemoveMultipleWhitespaces() {
056                return removeMultipleWhitespaces;
057        }
058        public void setRemoveMultipleWhitespaces(boolean removeMultipleWhitespaces) {
059                this.removeMultipleWhitespaces = removeMultipleWhitespaces;
060        }
061
062        public boolean isTrimIt() {
063                return trimIt;
064        }
065        public void setTrimIt(boolean trimIt) {
066                this.trimIt = trimIt;
067        }
068}
