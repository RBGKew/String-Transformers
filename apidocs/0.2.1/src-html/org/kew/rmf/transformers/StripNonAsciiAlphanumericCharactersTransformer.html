010package org.kew.rmf.transformers;
011
012/**
013 * This transformer strips non ASCII alphanumeric characters (A to Z, a to z, 0 to 9), replacing
014 * them with a space.
015 * <br/>
016 * Could often be combined with {@link NormaliseDiacriticalMarksTransformer} first.
017 *
018 * @see StripNonAlphabeticCharactersTransformer for retaining accented characters etc.
019 */
020public class StripNonAsciiAlphanumericCharactersTransformer extends RegexTransformer {
021
022        public StripNonAsciiAlphanumericCharactersTransformer() {
023                super.setPattern("[^A-Za-z0-9]");
024                super.setReplacement(" ");
025        }
026}
