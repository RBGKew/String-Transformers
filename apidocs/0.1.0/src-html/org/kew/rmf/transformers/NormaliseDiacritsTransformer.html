010package org.kew.rmf.transformers;
011
012import java.text.Normalizer;
013import java.util.HashMap;
014import java.util.Map;
015
016/**
017 * Converts a string to its normalised form using unicode's NFD form
018 * (http://unicode.org/reports/tr15/) adding an ASCII equivalent after any
019 * identified diacritical character) and then removes
020 * all non-ASCII characters from the string
021 *
022 * In addition to the characters defined by the unicode consortium the following
023 * characters are replaced: [[ADDITIONAL)_REPLACEMENTS]]
024 */
025public class NormaliseDiacritsTransformer implements Transformer {
026
027    @SuppressWarnings(value = { "serial" })
028    static Map<String,String> ADDITIONAL_REPLACEMENTS = new HashMap<String,String>() {{
029        put("Ø", "O");
030        put("ø", "o");
031        put("Ł", "L");
032        put("ł", "l");
033        put("—", "-"); // emdash to hyphen
034        put("–", "-"); // ndash to hyphen
035    }};
036
037    @Override
038    public String transform(String s) {
039        s = Normalizer.normalize(s, Normalizer.Form.NFD);
040        for (String key:ADDITIONAL_REPLACEMENTS.keySet()) {
041            s = s.replaceAll(key, ADDITIONAL_REPLACEMENTS.get(key));
042        }
043        return s.replaceAll("[^\\p{ASCII}]", "");
044    }
045
046}
