010package org.kew.rmf.transformers;
011
012import java.util.regex.Matcher;
013import java.util.regex.Pattern;
014
015/**
016 * A generic transformer that extracts all occurrences of a pattern (regEx)
017 * in a string
018 *
019 * It takes two optional parameters, `removeMultipleWhitespaces` (default true) and
020 * `trimIt` (default true)
021 */
022public class RegexExtractorTransformer extends RegexDefCollection implements Transformer {
023
024    protected String regex = "";
025    private boolean removeMultipleWhitespaces = true;
026    private boolean trimIt = true;
027
028    @Override
029    public String transform(String s) {
030        StringBuffer sb = new StringBuffer();
031        if (s != null){
032            if (this.removeMultipleWhitespaces) s = s.replaceAll("\\s+", " ");
033            if (this.trimIt) s = s.trim();
034                Pattern patt = Pattern.compile("(" + regex + ")");
035                Matcher m = patt.matcher(s);
036                while (m.find()){
037                        if (sb.length() > 0) sb.append(" ");
038                        sb.append(m.group());
039                }
040        }
041        return sb.toString();
042    }
043
044    public String getRegex() {
045                return regex;
046        }
047
048        public void setRegex(String regex) {
049                this.regex = regex;
050        }
051
052        public boolean isRemoveMultipleWhitespaces() {
053        return removeMultipleWhitespaces;
054    }
055
056    public void setRemoveMultipleWhitespaces(boolean removeMultipleWhitespaces) {
057        this.removeMultipleWhitespaces = removeMultipleWhitespaces;
058    }
059
060    public boolean isTrimIt() {
061        return trimIt;
062    }
063
064    public void setTrimIt(boolean trimIt) {
065        this.trimIt = trimIt;
066    }
067
068}
